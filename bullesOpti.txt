FONCTION bullesOpti(tableau t)

    variable indexMax <-- taille(t) - 1
    variable tableauTrié <-- faux
    TANT QUE tableauTrié == faux FAIRE
        tableauTrié <-- vrai

        POUR chaque index i allant de 0 à indexMax FAIRE
            SI t[i] > t[i+1] ALORS
                
                ECHANGER t[i] et t[i+1]
                tableauTrié <-- faux
                
                indexMax <-- i
            FIN SI
        FIN POUR
    
    FIN TANT QUE
    
    AFFICHER t
FIN FONCTION bullesOpti
